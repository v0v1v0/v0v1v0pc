Description

Numerical evaluation

Usage

N(x, digits = 15)

Arguments

x

caracas object
digits

Number of digits

Examples

if (has_sympy()) {
  n_2 <- as_sym("2")
  n_pi <- as_sym("pi", declare_symbols = FALSE)
  x <- sqrt(n_2) * n_pi
  x
  N(x)
  N(x, 5)
  N(x, 50)
  as.character(N(x, 50))
}
